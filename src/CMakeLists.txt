# The Flutter tooling requires that developers have CMake 3.10 or later
# installed. You should not increase this version, as doing so will cause
# the plugin to fail to compile for some customers of the plugin.
cmake_minimum_required(VERSION 3.10)

project(dnssec_proof_library VERSION 0.0.1 LANGUAGES C)

add_library(dnssec_proof SHARED
  "dnssec_proof.c"
)

set_target_properties(dnssec_proof PROPERTIES
  PUBLIC_HEADER dnssec_proof.h
  OUTPUT_NAME "dnssec_proof"
)

target_compile_definitions(dnssec_proof PUBLIC DART_SHARED_LIB)

if (ANDROID)
  # Support Android 15 16k page size
  target_link_options(dnssec_proof PRIVATE "-Wl,-z,max-page-size=16384")
endif()
